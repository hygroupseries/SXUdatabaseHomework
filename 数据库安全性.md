数据库安全性

```
CREATE USER zs WITH PASSWORD '123456';
```

```
CREATE ROLE R1;

GRANT SELECT, INSERT ON student TO R1;

GRANT DELETE ON sc TO R1;

GRANT UPDATE(grade) ON sc TO R1;
```

```
GRANT R1 TO zs;
```

```
SHOW GRANTS FOR R1;
SHOW GRANTS FOR zs;

```

```
REVOKE UPDATE(grade) ON sc FROM R1;
```

```
SHOW GRANTS FOR R1;
SHOW GRANTS FOR zs;
```
